\newacronym[description={Global System for Mobile Communications, 2nd generation mobile communication
standard, see \cite{gsm_book} for more information}]{GSM}{GSM}{global system for mobile communications}

\newacronym[description={Long Term Evolution, 4th generation mobile communication standard, see \cite{lte_paper} for more information}]{LTE}{LTE}{long term evolution}

\newacronym[description={File Transfer Protocol, built on top of TCP, see \cite{ftp_rfc} for more information}]{FTP}{FTP}{file transfer protocol}

\newacronym{PHY}{PHY}{physical layer protocol}

\newacronym{LPWAN}{LPWAN}{low power wide area network}

\newacronym[description={Internet of Things, it refers to the concept of a worldwide network of interconnected sensors}]{IoT}{IoT}{internet of things}

\newacronym[description={The Things Network, an open community of LoRaWAN gateways}]
{TTN}{TTN}{The Things Network}

\newacronym[description={LoRa Wide Area Network, it refers to an open network protocol built on top of LoRa}]{LoRaWAN}
{LoRaWAN}{LoRa wide area network}

\newacronym{CSMA}{CSMA}{carrier sense multiple access}

\newacronym{SYN}{SYN}{sync message}

\newacronym{SYN/ACK}{SYN/ACK}{sync acknowledge message}

\newacronym{ACK}{ACK}{acknowledge message}

\newacronym{ACKACK}{ACKACK}{double acknowledge message}
